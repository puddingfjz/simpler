cmake_minimum_required(VERSION 3.16.3)

# Build libtilefwk_backend_server.so from pypto_aicpu_interface.cpp
project(tile_fwk_server LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Set ASCEND paths
set(ASCEND_HOME_PATH $ENV{ASCEND_HOME_PATH})
if(NOT ASCEND_HOME_PATH)
    set(ASCEND_HOME_PATH "/usr/local/Ascend/cann-8.5.0")
endif()

message(STATUS "Building libtilefwk_backend_server.so")
message(STATUS "ASCEND_HOME_PATH: ${ASCEND_HOME_PATH}")

# Include directories
include_directories(
    ${CMAKE_CURRENT_SOURCE_DIR}/src
    ${ASCEND_HOME_PATH}/include
    ${ASCEND_HOME_PATH}/include/toolchain
    ${ASCEND_HOME_PATH}/pkg_inc/base
)

# Link directories
link_directories(
    ${ASCEND_HOME_PATH}/${CMAKE_SYSTEM_PROCESSOR}-linux/devlib/
    ${ASCEND_HOME_PATH}/lib64
)

# Compile pypto_aicpu_interface.cpp as shared library
add_library(tile_fwk_server SHARED
    src/pypto_aicpu_interface.cpp
)

target_compile_options(tile_fwk_server PRIVATE
    -Wall
    -Wextra
    -std=gnu++17
    -rdynamic
    -O3
    -shared
    -fPIC
    -D__DEVICE__
    -fno-common
)

target_link_libraries(tile_fwk_server PRIVATE
    -ldl
)

# Set output name
set_target_properties(tile_fwk_server PROPERTIES OUTPUT_NAME tilefwk_backend_server)

# Install to build directory
install(TARGETS tile_fwk_server LIBRARY DESTINATION build)
